// ========================================================================== // BLOC 7 — UNITÉS COMMERCIALES (Super Admin Work) // ========================================================================== let ucData = { dcis: [], membres: [], contrats: [] }; async function loadUC() { try { // Charger les profils DCI const [rDCIs, rMembres, rContrats] = await Promise.all([ sbGet('profiles', 'select=id,prenom,nom,role,region,profil_complet,profil_revocation_flag&role=eq.dci'), sbGet('profiles', 'select=id,prenom,nom,role,dci_parent_id,profil_complet,profil_revocation_flag,region&role=eq.user'), sbGet('contracts', 'select=id,created_by,montant,statut,type&statut=neq.Terminé'), ]); ucData.dcis = rDCIs || []; ucData.membres = rMembres || []; ucData.contrats = rContrats || []; // Stats globales const totalCA = ucData.contrats.reduce((s,c)=>s+(Number(c.montant)||0),0); const totalPrime = Math.round(totalCA * 0.03); document.getElementById('uc-stats').innerHTML = `
${ucData.dcis.length}
DCIs actifs
${ucData.membres.length}
Utilisateurs niveau 1
${totalCA.toLocaleString('fr-FR')} €
CA total équipes
${totalPrime.toLocaleString('fr-FR')} €
Primes cooptation totales
`; renderUC(); } catch(e) { console.error('loadUC:', e); } } function renderUC() { const filtreRegion = document.getElementById('uc-filtre-region')?.value || ''; const filtreStatut = document.getElementById('uc-filtre-statut')?.value || ''; let dcis = ucData.dcis; if (filtreRegion) dcis = dcis.filter(d => d.region === filtreRegion); if (filtreStatut === 'revocation') dcis = dcis.filter(d => d.profil_revocation_flag); if (filtreStatut === 'actif') dcis = dcis.filter(d => !d.profil_revocation_flag); document.getElementById('uc-count').textContent = dcis.length + ' DCI'; const tbody = document.getElementById('uc-tbody'); if (!dcis.length) { tbody.innerHTML = 'Aucun DCI correspondant.'; } else { tbody.innerHTML = dcis.map(dci => { const equipe = ucData.membres.filter(m => m.dci_parent_id === dci.id); const contrats = ucData.contrats.filter(c => equipe.some(m => m.id === c.created_by)); const ca = contrats.reduce((s,c)=>s+(Number(c.montant)||0),0); const prime = Math.round(ca * 0.03); const profil = dci.profil_complet ? '🟢' : '🔴'; const statut = dci.profil_revocation_flag ? '⚠ Révocation' : 'Actif'; return ` ${esc(dci.prenom||'')} ${esc(dci.nom||'')} ${esc(dci.region||'—')} ${equipe.length} membre${equipe.length>1?'s':''} ${equipe.length?'
'+equipe.map(m=>esc(m.prenom||'?')).join(', ')+'
':''} ${ca.toLocaleString('fr-FR')} € ${prime.toLocaleString('fr-FR')} € ${profil} ${statut} `; }).join(''); } // Orphelins (niveau 1 sans DCI) const orphelins = ucData.membres.filter(m => !m.dci_parent_id); document.getElementById('uc-orphelins-count').textContent = orphelins.length; const orphList = document.getElementById('uc-orphelins-list'); if (!orphelins.length) { orphList.innerHTML = '
✅ Tous les utilisateurs sont rattachés à un DCI.
'; } else { orphList.innerHTML = orphelins.map(m => `
${m.profil_complet?'🟢':'🔴'} ${esc(m.prenom||'')} ${esc(m.nom||'')} ${esc(m.region||'—')} ${m.profil_revocation_flag?'⚠ Révocation':''}
`).join(''); } } function esc(s){ return String(s).replace(/&/g,'&').replace(//g,'>'); }
